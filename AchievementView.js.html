define(function(require) {
  /**
   * Renders and handles events associated with an achievment in the 
   * achievement list on a pet's profile page.
   * Extends Backbone.View
   * @exports views/pet/AchievementView
   */
  var jquery            = require('libs/jquery'),
      underscore        = require('libs/underscore'),
      Backbone          = require('libs/backbone'),
      Achievement       = require('models/Achievement'),
      PictureUtils      = require('utils/PictureUtils'),
      achievementTpl    = require('text!/templates/achievement-tpl.html');
  AchievementView = Backbone.View.extend({

    initialize: function(options) {
      console.log(options);
      /** The achievement object this view uses to render its template properly. 
       * Note it is not an instance of Backbone.Model, but simply a JSON 
       * object. */
      this.achievement = options.achievement;

      /** Boolean indicating whether the pet has completed the achievement. */
      this.isComplete = options.isComplete;

      /** Boolean indication whether or not the pet has received no achievements
       * in the set. */
      this.hasNone = options.hasNone;

      /** Instance of PictureUtils.  Use for resizing of picture associated with
       * achievement. */
      this.picUtils = new PictureUtils({
        maxHeight:  80,
        maxWidth:   80
      });

      /** Integer value representing total percentage that the pet has earned
       * in the achievement. */
      this.percentageComplete;

      /** Steps completed in the achievement set. */
      this.stepsCompleted;

      /** String to be displayed explaining points earned/to be earned */
      this.pointsStr;

      /** The total number of points the set in which the achievement appears. */
      this.setPoints = options.setPoints;

      // Calculate correct percentage to display in progress bar
      if (this.hasNone == true) {
        this.percentageComplete = 0;
        this.pointsStr = "The first achievement in this set is worth " + 
        this.achievement.points + " points!";
        this.stepsComplete = 0;     
      }
      else if (this.isComplete == true) {
        this.stepsComplete = this.achievement.total;
        this.percentageComplete = 100;
        this.pointsStr = "You earned " + this.setPoints + " points for completing this set!";
      }
      else {
        this.stepsComplete = this.achievement.position;
        this.pointsStr = "You have earned " + options.pointsEarned + " points so far " + 
        "in this set.";
        this.percentageComplete = 
          Math.ceil(100 * (parseFloat(this.achievement.position) / 
          parseFloat(this.achievement.total)));
      }


    },

    className: 'achievement-el',

    render: function() {

      console.log(this.achievement);

      var tpl = _.template(achievementTpl, {
        achievement: this.achievement,
        percentageComplete: this.percentageComplete,
        pointsStr: this.pointsStr
      });
      $(this.el).html(tpl);

      this.picUtils.set({
        canvas: $(this.el).find('#resize-canvas').get(0),
        imgEl:  $(this.el).find('img').get(0)
      });

      this.picUtils.cropImage($(this.el).find('img').attr('src'));

      this.delegateEvents();

      return this;
    },

    events: {
      "mouseover.progress": "onProgressMouseover",
      "mouseleave .progress": "onProgressMouseleave",
    },

    onProgressMouseover: function(e) {
      var set = this.achievement.set;
      $(e.currentTarget).popover({
        animation: true,
        html: true,
        placement: 'top',
        trigger: 'hover',
        title: set.name,
        content: '<div>' + this.stepsComplete + " step(s) completed out of " + 
                  this.achievement.total +'.</div>' 
      });
      $(e.currentTarget).popover('show');
    },

    onProgressMouseleave: function(e) {
      $(e.currentTarget).popover('destroy');

    }
  });
  return AchievementView;
});
