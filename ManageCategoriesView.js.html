define(function(require) {
  /**
   * Renders and manages events associated with the Manage Categories modal
   * window on the administrative console page.
   * Extends Backbone.View
   * @exports views/admin/ManageCategoriesView
   */
  var jquery              = require('libs/jquery'),
      underscore          = require('libs/underscore'),
      Backbone            = require('libs/backbone'),
      Category            = require('models/InterestCategory'),
      CategoryList        = require('models/CategoryList'),
      CategoryListView    = require('views/admin/CategoryListView'),
      manageCategoriesTpl = require('text!/templates/admin/add-category-modal-tpl.html');

  ManageCategoriesView = Backbone.View.extend({

    initialize: function() {
      // If collection of categories have already been fetched from the server,
      // initialize view appropriately right now.
      if (this.collection.models.length > 0) {
        this.onCategoriesFetched();
      }
      this.model = Category.findOrCreate({id: -1});
      this.delegateListeners();
    },

    /** Class name of <div> DOM element into which this view will render its
     * template. */
    className: 'manage-categories-modal-el',

    /** Render view templae into this.el */
    render: function() {
      var tpl = _.template(manageCategoriesTpl, {});
      $(this.el).html(tpl);
      this.renderCategoryList();
      this.delegateEvents();
      return this;
    },

    /** Render the list of category managed by this view. */
    renderCategoryList: function() {
      if (this.categoryListView == undefined) return;

      $(this.el)
        .find('.category-list-col')
        .html(this.categoryListView.render().el);
    },

    /** Register DOM event listeners and callbacks. */
    events: {
      'click .close':   'hide',
      'keyup #new-category': 'checkForEnter',
      'click #confirm-add-category-btn': 'onAddNewCategory'
    },

    /** Show modal window associated with this view. */
    show: function() {
      console.log('adsfasdfasdf');
      $(this.el).find('.modal').modal('show');
    },

    /** Hide modal window associated with this view. */
    hide: function() {
      $(this.el).find('.modal').modal('hide');
    },

    /** Register model event listeners and callbacks. */
    delegateListeners: function() {
      this.collection.on('fetched', this.onCategoriesFetched, this);
      this.model.on('uploaded', this.onCategoryUpload, this);
    },

// Event callback functions. --------------------------------------------------/
    /** Callback to check for press of enter key while new interest description
     * input field has focus.  If pressed key is the enter key, triggers 
     * this.onAddNewCategory() */
    checkForEnter: function(e) {
      var keyCode = e.keyCode || e.which;
      if (keyCode == '13') {
        this.onAddNewCategory();
      }
    },

    onAddNewCategory: function(e) {
      var description = $(this.el).find('#new-category').val();
      this.model.set({description: description});
      this.model.post();
      return false;
    },

    onCategoryUpload: function() {
      this.collection.add(this.model);
      this.model = Category.findOrCreate({id: -1});
      $(this.el).find('#new-category').val('');
      this.delegateListeners();
    },

    onCategoriesFetched: function(e) {
      this.categoryListView =
        new CategoryListView(
          {
            parent: this,
            collection: this.collection
          }
        );
      this.renderCategoryList();
    },   
  });

  return ManageCategoriesView;

  });
