define(function(require) {
  /**
   *
   * Extends Backbone.Model
   * @exports models/Login
   */
  var jquery   			   	= require('libs/jquery'),
	  underscore 			= require('libs/underscore'),
	  Backbone 		     	= require('libs/backbone'),
	  ServerController 		= require('controllers/ServerController');

  Login = Backbone.Model.extend({
	  
	/**
	 * Initialize this model with a ServerController to be used later on
	 */  
    // Function called on model instantiation.
    initialize: function() {
      this.sc = new ServerController();
    },
    
    
    // Default values that the class is instantiated with.
    defaults: {
    },

    // Owner validation functions -------------------------------------------------/

    /**
     * Function called to validate the login input fields, called on each 
     * attempt to set a model attribute if the set method is also passed the
     * objection {validate: true}.
     */
    validate: function(attrs) {
      var errors = {};
      var hasError = false;
      
      if (attrs == undefined) attrs = this.attributes;

      if (attrs.username == '' || attrs.password == '') {
    	  hasError = true;
    	  errors.message = ('Please complete all fields');
      }

      if (hasError == true) {
        this.trigger('error', errors);
        return false;
      } else return true;

    },

// Network functionality ------------------------------------------------------/
    /**
     * Sends the login data to the server for authentication.
     */
    save: function() {
    	
      // Errors triggered in validation function.
      if (!this.validate()) return; 
      
      var postData = $.param(this.toJSON());
      var self = this;
      
      this.sc.post(
        '/login',
        postData,
        onSuccess,
        function() {}
      );
    
      // On a successful login request
      function onSuccess(data, status, req) {
        var parsedData;
        try {
          parsedData = $.parseJSON(data);
        } catch (e) {
          console.log(e);
          console.log(data);
          //window.location = '/500';
          return;
        }
        
        if (parsedData['loginSuccessful'] == false) {

            self.trigger('error', {message: 'Login failed'});

        } else {

          window.location = parsedData['redirectTo'];

        }
      }
    }
  });

  return Login;



});
