define(function(require) {
  /**
   * Renders list of pets on owner profile and manages associated events.
   * Extends Backbone.View
   * @exports views/owner/PetListView
   */
  var jquery              = require('libs/jquery'),
      underscore          = require('libs/underscore'),
      Backbone            = require('libs/backbone'),
      Pet                 = require('models/Pet'),
      Owner               = require('models/Owner'),
      PetListElementView  = require('views/owner/PetListElementView');

  PetListView = Backbone.View.extend({

    initialize: function(options) {
      console.log(options.owner);
      this.owner = options.owner;

      this.delegateListeners();
      /** Id of pet to be displayed as "active" in the pet list. */
      this.activePet = null;
    },

    /** Render View template in this.el */
    render: function() {

      if (this.views == undefined | null) return this; 

      $(this.el).html('');

      var container = document.createDocumentFragment();
      var self = this;
      this.views.forEach(function(plev) {
        container.appendChild(plev.render().el);
        if (plev.model.get('id') == self.activePet) {
          plev.setActive();
        } else {
          plev.setInactive();
        }
      });

      $(this.el).append(container);

      this.delegateEvents();

      return this;

    },

    /** Calss of div DOM element into which the template will be rendered. */
    className: 'pet-list-el',

    /** Define events listeners and callbacks. */
    events: {
      'click .pet-listing': 'redirect',
      'click #add-pet-btn': 'showAddPet'
    },

    delegateListeners: function() {
      this.owner.on('fetched', this.onOwnerFetch, this);
      if (this.collection != undefined | null) {
        this.collection.on('add', this.onAdd, this);
      }
    },

    /** 
     * Callback for click of pet in list of pets.  Redirect user to the pet's
     * profile page.
     */
    redirect: function(e) {
      router.navigate('/owner/' + this.model.get('id') + '/pets/' + id, true);
    },

    /**
     * Renders the pet with the given id in the list as the "active" pet.
     * @param {string} petId - The id of the pet to be displayed as "active"
     * in the pet list.
     */
    setActivePet: function(petId) {
      this.activePet = petId;
      this.render();
    },

    /**
     *
     *
     */
     removeAllActive: function() {
      this.activePet = null;
      this.render();
     },

    /**
     * Navigates to user to view from which they can add a pet to the owner 
     * profile page.
     */
    showAddPet: function(e) {
      console.log('fired');
      router.navigate('/owner/' + this.model.get('id') + '/pets/add', true);
    },

    onAdd: function() {
      console.log('pet list view on add triggered...');
    },

    onOwnerFetch: function() {
      var self = this;
      this.collection = this.owner.get('pets');

      /** Collection of subviews.  Instances of PetListElementView. */
      this.views = [];
      this.collection.models.forEach(function(pet) {
        var plev = new PetListElementView(
          {
            model: pet, 
            parent: self,
            ownerId: self.owner.get('id')
          }
        );
        self.views.push(plev);
      });
      this.delegateListeners();
      this.render();
    }
  });

  return PetListView;

});
