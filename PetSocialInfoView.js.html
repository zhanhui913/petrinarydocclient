define(function(require) {
  var jquery            = require('libs/jquery'),
    underscore          = require('libs/underscore'),
    Backbone            = require('libs/backbone'),
    AddInterestView     = require('views/pet/AddInterestView'),
    CarouselView        = require('views/pet/CarouselView'),
    ManagePicturesView  = require('views/pet/ManagePicturesView'), 
    ZoomPictureView     = require('views/pet/ZoomPictureView'),
    InterestListView    = require('views/pet/PetInterestList'),
    petSocialInfoTpl    = require('text!/templates/pet/pet-social-info-view-tpl.html');

  PetSocialInfoView = Backbone.View.extend({

    initialize: function(options) {
      this.model = options.model;
      this.carouselView = new CarouselView({parent: this});
      this.addInterestView = new AddInterestView({parent: this, pet: this.model});
      this.interestListView = new InterestListView({parent: this});
      this.registerListeners();
    },

    render: function() {

      this.carouselView.set({collection: this.model.pictures});
      this.addInterestView.set({pet: this.model});
      this.interestListView.set({collection: this.model.interests});

      var tpl = _.template(petSocialInfoTpl, {pet: this.model.toJSON()});
      // el passed in at instantiation
      $(this.el).html(tpl);

      $(this.el).find('#interests-header').html(this.addInterestView.render().el);

      $(this.el).find('#picture-center').html(this.carouselView.render().el);

      $(this.el).find('.interests-list').html(this.interestListView.render().el);

      return this;

    },

    el: $('<div>').addClass('content'),

    // Set the events that view listens for and corresponding actions.
    events: {
      'click #manage-pictures-btn':   'showManagePictures'
    },

    registerListeners: function() {
      this.on('zoom', this.showZoomPicture, this);
    },
// Event callback functions. --------------------------------------------------/

    /**
     * Called to instantiate, render and display the manage pictures window.
     * @param: [Event] e : the event object.
     */
    showManagePictures: function(e) {
      
      if (this.managePicturesView == undefined | null) {
        this.managePicturesView = 
          new ManagePicturesView(
            {
              collection: this.model.pictures,
              parent: this
            }
          );
        $('html').append(this.managePicturesView.render().el);
        this.managePicturesView.show();
      } else {
        this.managePicturesView.render();
        this.managePicturesView.show();
      }
    },

    /**
     * Launch zoomed picture view modal window.
     */
    showZoomPicture: function(picture) {
      console.log('shbow zoom picture firing...');
      if (this.zoomPictureView == undefined | null) {
        this.zoomPictureView = new ZoomPictureView({model: picture});
        $('html').append(this.zoomPictureView.render().el);
        this.zoomPictureView.show();
      } else {
        this.zoomPictureView.changeModel(picture);
        this.zoomPictureView.show();
      }

    }

  });

  return PetSocialInfoView;

});
